// Fonction pour ajouter une décoration function addDecoration(imageName) { const img = document.createElement('img'); img.src = 'decorations/' + imageName; img.className = 'decor'; img.style.position = 'absolute'; img.style.top = '100px'; img.style.left = '100px'; img.style.width = '80px'; img.draggable = true; makeDraggable(img); document.body.appendChild(img); } // Fonction pour ajouter du texte décoratif function addText(textContent) { const text = document.createElement('div'); text.className = 'decor-text'; text.innerText = textContent; text.style.position = 'absolute'; text.style.top = '150px'; text.style.left = '150px'; text.style.fontSize = '24px'; text.style.color = 'green'; text.style.fontWeight = 'bold'; text.draggable = true; makeDraggable(text); document.body.appendChild(text); } // Rendre les éléments déplaçables function makeDraggable(element) { let offsetX, offsetY; element.addEventListener('dragstart', function(e) { offsetX = e.offsetX; offsetY = e.offsetY; }); document.body.addEventListener('dragover', function(e) { e.preventDefault(); }); document.body.addEventListener('drop', function(e) { e.preventDefault(); element.style.left = (e.pageX - offsetX) + 'px'; element.style.top = (e.pageY - offsetY) + 'px'; }); }
